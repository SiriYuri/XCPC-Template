template<class Info>
struct SegmentTree {
    int n;                        
    std::vector<Info> info;

    SegmentTree(int len, Info v_ = Info()) {
        init(len, v_);
    }

    SegmentTree(const std::vector<Info>& a) {
        init(a);
    }

    void init(int len, Info v_ = Info()) {
        std::vector<Info> a(len, v_);
        init(a);
    }

    void init(const std::vector<Info>& a) {
        int len = (int)a.size();
        n = len - 1;                   
        info.assign(4 * (n + 1) + 5, Info());

        std::function<void(int, int, int)> build = [&](int p, int l, int r) {
            if (l == r) {
                info[p] = a[l];
                return;
            }
            int m = (l + r) / 2;
            build(p * 2, l, m);
            build(p * 2 + 1, m + 1, r);
            pull(p);
        };

        build(1, 0, n);
    }

    void pull(int p) {
        info[p] = info[p * 2] + info[p * 2 + 1];
    }

    void modify(int p, int l, int r, int x, const Info &v) {
        if (l == r) {
            info[p] = v;
            return;
        }
        int m = (l + r) / 2;
        if (x <= m) {
            modify(p * 2, l, m, x, v);
        } else {
            modify(p * 2 + 1, m + 1, r, x, v);
        }
        pull(p);
    }

    void modify(int x, const Info &v) {
        modify(1, 0, n, x, v);
    }

    Info rangequery(int p, int l, int r, int x, int y) {
        if (r < x || l > y) {
            return Info();
        }
        if (x <= l && r <= y) {
            return info[p];
        }
        int m = (l + r) / 2;
        return rangequery(p * 2, l, m, x, y) + rangequery(p * 2 + 1, m + 1, r, x, y);
    }

    Info rangequery(int l, int r) {
        if (l > r) return Info();
        return rangequery(1, 0, n, l, r);
    }
};

int inf = std::numeric_limits<int>::max() / 2;

struct Info {
    i64 sum = 0;
    int max = -inf;
    int min = inf;
};

Info operator + (const Info &a, const Info &b) {
    return {a.sum + b.sum, std::max(a.max, b.max), std::min(a.min, b.min)};
}
