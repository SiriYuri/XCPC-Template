struct DSU {
    int n;
    std::vector<int> fa, siz, cnt;

    DSU (int _n) : n(_n), fa(n + 1), siz(n + 1, 1), cnt(n + 1) {
        std::iota(fa.begin(), fa.end(), 0);
    }

    int find(int x) {
        return fa[x] == x ? x : fa[x] = find(fa[x]);
    }

    bool same(int u, int v) {
        return find(u) == find(v);
    }

    void merge(int u, int v) {
        u = find(u);
        v = find(v);
        if (u == v) {
            cnt[v]++;
        } else {
            fa[u] = v;
            siz[v] += siz[u];
            cnt[v] += cnt[u] + 1;
        }
    }

    int size(int x) {
        return siz[find(x)];
    }

    int cntEdge(int x) {
        return cnt[find(x)];
    }
};
