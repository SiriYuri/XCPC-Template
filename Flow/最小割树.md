$O(n^3m)$ 预处理, $O(log)$ 查询两点间最小割  
要求无向图

template<typename T>
struct MaxFlow {
    struct _Edge {
        int to;
        T cap;

        _Edge(int to, T cap) : to(to), cap(cap) {}
    };

    int n;
    std::vector<_Edge> e;
    std::vector<std::vector<int>> g;
    std::vector<int> cur;
    std::vector<int> h;

    MaxFlow() {}

    MaxFlow(int n) {
        init(n);
    }

    void init(int _n) {
        this->n = _n;
        e.clear();
        g.assign(n + 1, {});
        cur.resize(n + 1);
        h.resize(n + 1);
    }

    bool bfs(int s, int t) {
        h.assign(n + 1, -1);  
        std::queue<int> que;
        h[s] = 0;        
        que.push(s);     
        while (!que.empty()) {
            const int u = que.front(); 
            que.pop();
            for (int i : g[u]) {
                auto [v, c] = e[i];  
                if (c > 0 && h[v] == -1) {
                    h[v] = h[u] + 1; 
                    if (v == t) {    
                        return true;
                    }
                    que.push(v);  
                }
            }
        }
        return false;  
    }

    T dfs(int u, int t, T f) {
        if (u == t) return f;
        T r = f; 
        for (int &i = cur[u]; i < int(g[u].size()); ++i) {
            const int j = g[u][i];
            auto [v, c] = e[j];
            if (c > 0 && h[v] == h[u] + 1) {
                auto a = dfs(v, t, std::min(r, c));
                if (a == 0) {
                    h[v] = -1; 
                } else {
                    e[j].cap -= a;     
                    e[j ^ 1].cap += a; 
                    r -= a; 
                    if (r == 0) {  
                        return f;
                    }
                }
            }
        }
        return f - r;  
    }

    void addEdge(int u, int v, T c) {
        g[u].push_back(e.size());  
        e.emplace_back(v, c);    
        g[v].push_back(e.size()); 
        e.emplace_back(u, 0);    
    }

    T flow(int s, int t) {
        T ans = 0; 
        while (bfs(s, t)) {
            cur.assign(n + 1, 0);  
            ans += dfs(s, t, std::numeric_limits<T>::max());
        }
        return ans;
    }
};

template<typename T>
struct GomoryHuTree {
    int n;
    MaxFlow<T> mf;
    std::vector<T> initial_cap; 

    struct Edge {
        int to;
        T w;
    };
    std::vector<std::vector<Edge>> tree;

    GomoryHuTree(int n) : n(n), mf(n), tree(n + 1) {}

    void addEdge(int u, int v, T w) {
        mf.addEdge(u, v, w);
        mf.addEdge(v, u, w);
    }

    void build() {
        initial_cap.resize(mf.e.size());
        for (size_t i = 0; i < mf.e.size(); ++i) {
            initial_cap[i] = mf.e[i].cap;
        }

        std::vector<int> nodes;
        for (int i = 0; i <= n; ++i) {
            nodes.push_back(i);
        }

        solve(nodes);
        build_lca();
    }

private:
    void solve(std::vector<int>& nodes) {
        if (nodes.size() <= 1) return;
        int s = nodes[0], t = nodes[1];

        for (size_t i = 0; i < mf.e.size(); ++i) {
            mf.e[i].cap = initial_cap[i];
        }

        T cut_val = mf.flow(s, t);
        
        tree[s].push_back({t, cut_val});
        tree[t].push_back({s, cut_val});

        std::vector<int> left, right;
        for (int u : nodes) {
            if (mf.h[u] != -1) left.push_back(u);
            else right.push_back(u);
        }

        solve(left);
        solve(right);
    }

    int LOG;
    std::vector<std::vector<int>> up;
    std::vector<std::vector<T>> min_w;
    std::vector<int> depth;

    void build_lca() {
        LOG = std::__lg(n + 1) + 2;
        up.assign(n + 1, std::vector<int>(LOG, 0));
        min_w.assign(n + 1, std::vector<T>(LOG, std::numeric_limits<T>::max()));
        depth.assign(n + 1, 0);

        dfs(0, 0, std::numeric_limits<T>::max());
    }

    void dfs(int u, int p, T w) {
        depth[u] = depth[p] + 1;
        up[u][0] = p;
        min_w[u][0] = w;

        for (int i = 1; i < LOG; ++i) {
            up[u][i] = up[up[u][i - 1]][i - 1];
            min_w[u][i] = std::min(min_w[u][i - 1], min_w[up[u][i - 1]][i - 1]);
        }

        for (auto& edge : tree[u]) {
            int v = edge.to;
            if (v != p) {
                dfs(v, u, edge.w);
            }
        }
    }

public:
    T query(int u, int v) {
        if (depth[u] < depth[v]) std::swap(u, v);
        T ans = std::numeric_limits<T>::max();

        for (int i = LOG - 1; i >= 0; --i) {
            if (depth[u] - (1 << i) >= depth[v]) {
                ans = std::min(ans, min_w[u][i]);
                u = up[u][i];
            }
        }

        if (u == v) return ans;

        for (int i = LOG - 1; i >= 0; --i) {
            if (up[u][i] != up[v][i]) {
                ans = std::min({ans, min_w[u][i], min_w[v][i]});
                u = up[u][i];
                v = up[v][i];
            }
        }

        ans = std::min({ans, min_w[u][0], min_w[v][0]});
        return ans;
    }
};
