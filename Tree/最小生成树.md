Prime   
维护连通块内点到连通块外点最小距离
```cpp
auto prim = [&] () -> i64 {
    std::vector<i64> mind(n + 1, inf<i64>), vis(n + 1);
    std::priority_queue<std::pair<i64, i64>, std::vector<std::pair<i64, i64>>, std::greater<>> pq;
    i64 res = 0, cnt = 0;
    mind[1] = 0;
    pq.emplace(0, 1);
    while (!pq.empty()) {
        auto [w, u] = pq.top();
        pq.pop();
        if (vis[u]) {
            continue;
        }
        vis[u] = 1;
        res += w;
        cnt++;
        for (auto [v, d] : edge[u]) {
            if (!vis[v] && d < mind[v]) {
                mind[v] = d;
                pq.emplace(mind[v], v);
            }
        }
    }
    return cnt == n ? res : -1; 
};
```
Borůvka   
每一轮都寻找这个连通块到连通块外的最小边权   
一般都是数据结构优化
```cpp
auto boruvka = [&] () -> i64 {
    DSU dsu(n);
    i64 res = 0, cnt = n; 
    while (cnt > 1) {
        std::vector<int> minw(n + 1, -1);
        for (int i = 0; i < m; i++) {
            auto [u, v, w] = e[i];
            if (dsu.same(u, v)) {
                continue;
            }
            int fu = dsu.find(u);
            int fv = dsu.find(v);
            if (minw[fu] == -1 || w < e[minw[fu]][2]) {
                minw[fu] = i;
            }
            if (minw[fv] == -1 || w < e[minw[fv]][2]) {
                minw[fv] = i;
            }
        }
        bool upd = 0;
        for (int i = 1; i <= n; i++) {
            if (dsu.find(i) == i && minw[i] != -1) {
                int idx = minw[i];
                auto [u, v, w] = e[idx];
                int fu = dsu.find(u);
                int fv = dsu.find(v);
                if (fu != fv) {
                    dsu.merge(u, v);
                    cnt--;
                    res += w;
                    upd = 1;
                }
            }
        }
        if (!upd) {
            break;
        }
    }
    return cnt == 1 ? res : -1;
};  
```
