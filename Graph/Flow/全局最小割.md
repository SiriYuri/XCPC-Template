$O(n^3)$
```cpp
struct StoerWagner {
    int n;
    std::vector<std::vector<i64>> g;
    std::vector<i64> w;   
    std::vector<int> id;       
    std::vector<bool> vis;     

    const i64 INF = 1e18;

    StoerWagner(int n) : n(n), g(n + 1, std::vector<i64>(n + 1, 0)), 
                         w(n + 1), id(n + 1), vis(n + 1) {
        for (int i = 1; i <= n; ++i) id[i] = i;
    }

    void addEdge(int u, int v, i64 weight) {
        g[u][v] += weight;
        g[v][u] += weight;
    }

    i64 min_cut() {
        if (n < 2) return 0;
        i64 min_val = INF;
        for (int i = n; i > 1; --i) {
            std::fill(w.begin(), w.end(), 0);
            std::fill(vis.begin(), vis.end(), false);
            int prev = 0, curr = 0;
            for (int j = 1; j <= i; ++j) {
                int next = -1;
                for (int k = 1; k <= i; ++k) {
                    if (!vis[id[k]] && (next == -1 || w[id[k]] > w[id[next]])) {
                        next = k;
                    }
                }
                vis[id[next]] = true;
                if (j == i) { 
                    min_val = std::min(min_val, w[id[next]]);
                    for (int k = 1; k <= i; ++k) {
                        g[id[prev]][id[k]] += g[id[next]][id[k]];
                        g[id[k]][id[prev]] = g[id[prev]][id[k]];
                    }
                    id[next] = id[i];
                } else {
                    prev = next;
                    for (int k = 1; k <= i; ++k) {
                        if (!vis[id[k]]) w[id[k]] += g[id[next]][id[k]];
                    }
                }
            }
        }
        return min_val == INF ? 0 : min_val;
    }
};
```
