            if (dis[v] > dis[u] + w) {
                dis[v] = dis[u] + w;
                cnt[v] = cnt[u] + 1;
                if (cnt[v] >= n) {
                    return true;
                }
                if (!vis[v]) {
                    vis[v] = 1;
                    que.emplace(v);
                }
            }
        }
    }
    return false;
};

auto negitaveCircle = [&] () -> std::vector<int> {
    std::queue<int> que;
    std::vector<int> dis(n + 1), cnt(n + 1), vis(n + 1, 1), pre(n + 1, 0); 
    for (int i = 1; i <= n; i++) {
        que.emplace(i);
    }
    while(!que.empty()) {
        int u = que.front();
        que.pop();
        vis[u] = 0;
        for (auto &[v, w] : edge[u]) {
            if (dis[v] > dis[u] + w) {
                dis[v] = dis[u] + w;
                pre[v] = u;             
                cnt[v] = cnt[u] + 1;
                if (cnt[v] >= n) {
                    int cur = v;
                    for (int i = 0; i < n; i++) {
                        cur = pre[cur];
                    }
                    std::vector<int> cycle;
                    int start = cur;
                    do {
                        cycle.emplace_back(cur);
                        cur = pre[cur];
                    } while (cur != start);
                    cycle.emplace_back(start);
                    std::reverse(cycle.begin(), cycle.end());
                    return cycle;
                }
                if (!vis[v]) {
                    vis[v] = 1;
                    que.emplace(v);
                }
            }
        }
    }
    return {}; 
};
