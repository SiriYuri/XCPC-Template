auto dfs = [&] (this auto &&self, int p, int fa) -> void {
    int mx1 = 0, mx2 = 0;
    for (auto c : edge[p]) {
        if (c != fa) {
            self(c, p);
            if (mx1 < dis[c] + 1) {
                mx2 = mx1;
                mx1 = dis[c] + 1;
            } else if (mx2 < dis[c] + 1) {
                mx2 = dis[c] + 1;
            }
        }
    }
    dis[p] = mx1;
    ans = std::max(ans, mx1 + mx2);
};
