dijkstra   
返回 [最短路, 前驱节点, 路径数]
```cpp
auto dijk = [&] (int root) -> auto {
    std::vector<i64> dis(n + 1, inf<i64>), pre(n + 1, -1), cnt(n + 1);
    std::priority_queue<std::pair<i64, i64>, std::vector<std::pair<i64, i64>>, std::greater<std::pair<i64, i64>>> pq;
    dis[root] = 0;
    cnt[root] = 1;
    pq.emplace(0, root);
    while (!pq.empty()) {
        auto [d, u] = pq.top();
        pq.pop();
        if (d != dis[u]) {
            continue;
        }
        for (auto [to, w] : edge[u]) {
            if (dis[to] > dis[u] + w) {
                dis[to] = dis[u] + w;
                pre[to] = u;
                cnt[to] = cnt[u];
                pq.emplace(dis[to], to);
            } else if (dis[to] == dis[u] + w) {
                cnt[to] += cnt[u];
            }
        }
    }
    return std::make_tuple(dis, pre, cnt);
};
```
spfa   
带负权的最短路   
注意不满足拓扑序, 不能之间转移的时候计算路径数, 要先求出dis数组, 再重构最短路dag再转移
```cpp
auto spfa = [&] (int root) {
    std::vector<i64> dis(n + 1, inf<i64>), inq(n + 1); 
    std::queue<int> que;
    dis[root] = 0;
    que.emplace(root);
    inq[root] = true;
    while (!q.empty()) {
        int u = q.front();
        q.pop();
        inq[u] = 0; 
        for (auto [to, w] : edge[u]) {
            if (dis[to] > dis[u] + w) {
                dis[to] = dis[u] + w;
                if (!inq[to]) {
                    q.push(to);
                    inq[to] = true;
                }
            }
        }
    }
    return dis; 
};
```

Johnson   
带负权的全源最短路   
$O(NM + NMlogN)$
```cpp
struct Johnson {
    int n;
    struct Edge { 
        int to; i64 w; 
    };
    std::vector<std::vector<Edge>> adj;
    std::vector<i64> h;

    Johnson(int n) : n(n), adj(n + 1), h(n + 1, 0) {}

    void add(int u, int v, i64 w) {
        adj[u].emplace_back(Edge{v, w});
    }

    bool check() {
        std::vector<int> cnt(n + 1, 0);
        std::vector<bool> inq(n + 1, true);
        std::queue<int> q;
        for (int i = 1; i <= n; i++) {
            q.emplace(i);
        }
        while (!q.empty()) {
            int u = q.front(); q.pop();
            inq[u] = false;
            for (auto& e : adj[u]) {
                if (h[e.to] > h[u] + e.w) {
                    h[e.to] = h[u] + e.w;
                    if (!inq[e.to]) {
                        if (++cnt[e.to] >= n) return false;
                        q.emplace(e.to);
                        inq[e.to] = true;
                    }
                }
            }
        }
        return true;
    }

    std::vector<std::vector<i64>> work() {
        if (!check()) return {}; 
        std::vector<std::vector<i64>> res(n + 1, std::vector<i64>(n + 1, inf<i64>));
        for (int i = 1; i <= n; i++) {
            std::vector<i64> dis(n + 1, inf<i64>);
            std::priority_queue<std::pair<i64, int>, std::vector<std::pair<i64, int>>, std::greater<>> pq;
            dis[i] = 0;
            pq.emplace(0, i);
            while (!pq.empty()) {
                auto [d, u] = pq.top(); pq.pop();
                if (d > dis[u]) continue;
                for (auto& e : adj[u]) {
                    if (dis[e.to] > dis[u] + e.w + h[u] - h[e.to]) {
                        dis[e.to] = dis[u] + e.w + h[u] - h[e.to];
                        pq.emplace(dis[e.to], e.to);
                    }
                }
            }
            for (int j = 1; j <= n; j++) {
                if (dis[j] != inf<i64>) {
                    res[i][j] = dis[j] - h[i] + h[j];
                }
            }
        }
        return res;
    }
};
```
