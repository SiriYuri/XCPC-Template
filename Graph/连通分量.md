强连通分量
```cpp
struct SCC {
    int n, tot, cnt, cur;
    std::vector<int> dfn, low, bel, stk;
    std::vector<std::vector<int>> adj;

    SCC(int _n) : n(_n), tot(0), cnt(0), cur(-1) adj(n + 1, std::vector<int> ()), bel(n + 1), dfn(n + 1), low(n + 1) {}

    void add(int u, int v) {
        adj[u].emplace_back(v);
    }

    void Tarjan(int p) {
        dfn[p] = low[p] = ++tot;
        stk.emplace_back(p);
        for (auto &c : adj[p]) {
            if (!dfn[c]) {
                Tarjan(c);
                low[p] = std::min(low[p], low[c]);
            } else if (!bel[c]) {
                low[p] = std::min(low[p], dfn[c]);
            }
        }
        if (dfn[p] == low[p]) {
            ++cnt;
            do {
                cur = stk.back();
                bel[cur] = cnt;
                stk.pop_back();
            } while (cur != p);
        }
    }

    auto work() { // 新图点数, 新图, 原图点的归属, 新图点的size
        for (int i = 1; i <= n; i++) {
            if (!dfn[i]) {
                Tarjan(i);
            }
        }
        std::vector<int> siz(cnt + 1);
        std::vector<std::vector<int>> edge(n + 1);
        for (int i = 1; i <= n; i++) {
            siz[bel[i]]++;
            for (auto &c : adj[i]) {
                int x = bel[i], y = bel[c];
                if (x ^ y) {
                    edge[x].emplace_back(y);
                }
            }
        }
        return std::make_tuple(cnt, edge, bel, siz);
    }
};
```

边双   
边双中任意两点, 一定存在至少两条不相交的路径(允许共用点)
```cpp
struct ECC {
    int n, tot, cnt;
    std::vector<int> dfn, low, id, stk;
    std::vector<std::vector<int>> edge, adj;
    std::set<std::array<int, 2>> bri;
  
    ECC(int _n) : n(_n), tot(0), cnt(0), dfn(n + 1, -1), id(n + 1, -1), low(n + 1), edge(n + 1) {}

  
    void add(int u, int v) {
        edge[u].emplace_back(v);
        edge[v].emplace_back(u);
    }

  
    void Tarjan(int p, int fa) {
        dfn[p] = low[p] = ++tot;
        stk.emplace_back(p);
        for (auto c : edge[p]) {
            if (c == fa) {
                continue;
            }
            if (dfn[c] == -1) {
                bri.insert({p, c});
                Tarjan(c, p);
                low[p] = std::min(low[p], low[c]);
            } else if (id[c] == -1 && dfn[c] < dfn[p]) {
                bri.insert({c, p});
                low[p] = std::min(low[p], dfn[c]);
            }
        }
        if (dfn[p] == low[p]) {
            int pre;
            cnt++;
            do {
                pre = stk.back();
                id[pre] = cnt;
                stk.pop_back();
            } while(pre != p);
        }
    }


    auto work() { // 新图点数, 新图, 原图点的归属, 新图点的size
        for (int i = 1; i <= n; i++) {
            if (dfn[i] == -1) {
                Tarjan(i, 0);
            }
        }
        std::vector<int> siz(cnt + 1);
        adj.resize(cnt + 1);
        for (int i = 1; i <= n; i++) {
            siz[id[i]]++;
            for (auto c : edge[i]) {
                int x = id[i], y = id[c];
                if (x != y) {
                    adj[x].emplace_back(y);
                }
            }
        }
        return std::tuple{cnt, adj, id, siz};
    }
};
```

```cpp
struct VCC {
    int n, tot, cnt;
    std::vector<int> is_cut, dfn, low, stk;
    std::vector<std::vector<int>> edge, vcc;

    VCC(int _n) : n(_n), tot(0), cnt(0), is_cut(n + 1), dfn(n + 1), low(n + 1), edge(n + 1), vcc(n + 1) {
        stk.clear();
    }

    void add(int u, int v) {
        edge[u].emplace_back(v);
        edge[v].emplace_back(u);
    }

    void Tarjan (int p, int rt) {
        low[p] = dfn[p] = ++tot;
        stk.emplace_back(p);
        if (p == rt && !edge[p].size()) {
            ++cnt;
            vcc[cnt].emplace_back(p);
            return ;
        }
        int f = 0;
        for (auto c : edge[p]) {
            if (!dfn[c]) {
                Tarjan(c, rt);
                low[p] = std::min(low[p], low[c]);
                if (dfn[p] <= low[c]) {
                    f++;
                    if (p != rt || f > 1) {
                        is_cut[p] = 1;
                    }
                    int cur;
                    ++cnt;
                    do {
                        cur = stk.back();
                        vcc[cnt].emplace_back(cur);
                        stk.pop_back();
                    } while(cur != c);
                    vcc[cnt].emplace_back(p);
                }
            } else {
                low[p] = std::min(low[p], dfn[c]);
            }
        }
    }

    void work() {
        for (int i = 1; i <= n; i++) {
            if (!dfn[i]) {
                Tarjan(i, i);
            }
        }
    }
};
```
